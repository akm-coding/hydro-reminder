```typescript
/**
 * lib/hydration.ts
 * Pure hydration math — no React, no Supabase dependencies.
 * All amounts stored in ml; convert for display only.
 */

/** Convert ml to oz (rounded to 1 decimal place) */
export const mlToOz = (ml: number): number =>
  parseFloat((ml / 29.5735).toFixed(1));

/** Convert oz to ml (rounded to nearest integer) */
export const ozToMl = (oz: number): number =>
  Math.round(oz * 29.5735);

/**
 * Suggest a daily goal in ml based on body weight in kilograms.
 * Formula: weight (kg) × 33 ml/kg
 * Clamped to minimum 1000ml and maximum 5000ml.
 */
export const suggestGoalMl = (weightKg: number): number => {
  const suggested = Math.round(weightKg * 33);
  return Math.max(1000, Math.min(5000, suggested));
};

/**
 * Returns true when the user has met or exceeded their daily goal.
 * Single canonical definition — use this everywhere, no inline comparisons.
 */
export const isGoalMet = (totalMl: number, goalMl: number): boolean =>
  totalMl >= goalMl;

/**
 * Format an amount for display in the user's preferred unit.
 * ml: shown as integer (e.g. "500 ml")
 * oz: shown to 1 decimal (e.g. "16.9 oz")
 */
export const formatAmount = (
  amountMl: number,
  unit: 'ml' | 'oz'
): string => {
  if (unit === 'oz') {
    return `${mlToOz(amountMl)} oz`;
  }
  return `${amountMl} ml`;
};

/**
 * Calculate goal progress as a percentage (0–100, capped at 100).
 */
export const goalProgressPercent = (totalMl: number, goalMl: number): number =>
  Math.min(100, Math.round((totalMl / goalMl) * 100));
```

```typescript
import { useState } from 'react';
import {
  View, Text, TextInput, TouchableOpacity, StyleSheet,
  KeyboardAvoidingView, Platform, ScrollView,
} from 'react-native';
import { useRouter } from 'expo-router';
import { suggestGoalMl, formatAmount } from '@/lib/hydration';

export default function OnboardingGoalScreen() {
  const router = useRouter();
  const [weightInput, setWeightInput] = useState('');
  const [goalInput, setGoalInput] = useState('');
  const [weightError, setWeightError] = useState('');
  const [goalError, setGoalError] = useState('');

  const handleWeightChange = (text: string) => {
    setWeightInput(text);
    setWeightError('');
    const kg = parseFloat(text);
    if (!isNaN(kg) && kg > 0) {
      setGoalInput(String(suggestGoalMl(kg)));
    }
  };

  const handleNext = () => {
    let valid = true;
    const kg = parseFloat(weightInput);
    if (weightInput.trim() === '' || isNaN(kg) || kg <= 0 || kg > 300) {
      setWeightError('Enter a valid body weight (1–300 kg)');
      valid = false;
    }
    const goal = parseInt(goalInput, 10);
    if (!goalInput.trim() || isNaN(goal) || goal < 500 || goal > 6000) {
      setGoalError('Enter a goal between 500 and 6000 ml');
      valid = false;
    }
    if (!valid) return;

    router.push({ pathname: '/(onboarding)/unit', params: { goalMl: goal } });
  };

  const kg = parseFloat(weightInput);
  const suggestedLabel =
    weightInput && !isNaN(kg) && kg > 0
      ? `Suggested: ${formatAmount(suggestGoalMl(kg), 'ml')}`
      : null;

  return (
    <KeyboardAvoidingView
      style={styles.container}
      behavior={Platform.OS === 'ios' ? 'padding' : 'height'}
    >
      <ScrollView contentContainerStyle={styles.scroll} keyboardShouldPersistTaps="handled">
        <Text style={styles.step}>Step 1 of 2</Text>
        <Text style={styles.title}>What's your daily water goal?</Text>
        <Text style={styles.subtitle}>
          We'll suggest a goal based on your body weight. You can adjust it to anything you like.
        </Text>

        <Text style={styles.label}>Body weight (kg)</Text>
        <TextInput
          style={[styles.input, weightError ? styles.inputError : null]}
          placeholder="e.g. 70"
          keyboardType="decimal-pad"
          value={weightInput}
          onChangeText={handleWeightChange}
          maxLength={5}
        />
        {weightError ? <Text style={styles.fieldError}>{weightError}</Text> : null}

        <Text style={styles.label}>
          Daily goal (ml){suggestedLabel ? ` — ${suggestedLabel}` : ''}
        </Text>
        <TextInput
          style={[styles.input, goalError ? styles.inputError : null]}
          placeholder="e.g. 2310"
          keyboardType="number-pad"
          value={goalInput}
          onChangeText={(text) => { setGoalInput(text); setGoalError(''); }}
          maxLength={5}
        />
        {goalError ? <Text style={styles.fieldError}>{goalError}</Text> : null}
        <Text style={styles.hint}>Change the value above to set a custom goal.</Text>

        <TouchableOpacity style={styles.button} onPress={handleNext}>
          <Text style={styles.buttonText}>Next</Text>
        </TouchableOpacity>
      </ScrollView>
    </KeyboardAvoidingView>
  );
}
```

```typescript
import { useState } from 'react';
import {
  View, Text, TouchableOpacity, StyleSheet, ActivityIndicator,
} from 'react-native';
import { useLocalSearchParams, useRouter } from 'expo-router';
import { supabase } from '@/lib/supabase';
import { useAuth } from '@/hooks/use-auth';
import { formatAmount } from '@/lib/hydration';

export default function OnboardingUnitScreen() {
  const { goalMl } = useLocalSearchParams<{ goalMl: string }>();
  const { user } = useAuth();
  const router = useRouter();
  const [selectedUnit, setSelectedUnit] = useState<'ml' | 'oz'>('ml');
  const [saving, setSaving] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const goal = parseInt(goalMl ?? '2000', 10);

  const handleFinish = async () => {
    if (!user) return;
    setSaving(true);
    setError(null);

    const { error: updateError } = await supabase
      .from('profiles')
      .update({
        daily_goal_ml: goal,
        unit_pref: selectedUnit,
        onboarding_complete: true,
        updated_at: new Date().toISOString(),
      })
      .eq('id', user.id);

    setSaving(false);

    if (updateError) {
      setError(updateError.message);
      return;
    }

    router.replace('/(tabs)');
  };

  return (
    <View style={styles.container}>
      <Text style={styles.step}>Step 2 of 2</Text>
      <Text style={styles.title}>Choose your unit</Text>
      <Text style={styles.subtitle}>
        Your goal: {formatAmount(goal, selectedUnit)}
      </Text>

      <View style={styles.options}>
        <TouchableOpacity
          style={[styles.option, selectedUnit === 'ml' && styles.optionSelected]}
          onPress={() => setSelectedUnit('ml')}
        >
          <Text style={[styles.optionLabel, selectedUnit === 'ml' && styles.optionLabelSelected]}>ml</Text>
          <Text style={[styles.optionSub, selectedUnit === 'ml' && styles.optionSubSelected]}>Millilitres</Text>
        </TouchableOpacity>

        <TouchableOpacity
          style={[styles.option, selectedUnit === 'oz' && styles.optionSelected]}
          onPress={() => setSelectedUnit('oz')}
        >
          <Text style={[styles.optionLabel, selectedUnit === 'oz' && styles.optionLabelSelected]}>oz</Text>
          <Text style={[styles.optionSub, selectedUnit === 'oz' && styles.optionSubSelected]}>Fluid ounces</Text>
        </TouchableOpacity>
      </View>

      <Text style={styles.goalDisplay}>
        Your daily goal: {formatAmount(goal, selectedUnit)}
      </Text>

      {error ? <Text style={styles.error}>{error}</Text> : null}

      <TouchableOpacity
        style={[styles.button, saving && styles.buttonDisabled]}
        onPress={handleFinish}
        disabled={saving}
      >
        {saving
          ? <ActivityIndicator color="#fff" />
          : <Text style={styles.buttonText}>Start Tracking</Text>
        }
      </TouchableOpacity>
    </View>
  );
}
```

```typescript
function AuthGate() {
  const { session, loading, onboardingComplete } = useAuthContext();
  const router = useRouter();
  const segments = useSegments();

  useEffect(() => {
    if (loading || (session && onboardingComplete === null)) return;

    const inAuthGroup = segments[0] === '(auth)';
    const inOnboardingGroup = segments[0] === '(onboarding)';

    if (!session && !inAuthGroup) {
      router.replace('/(auth)/sign-in');
    } else if (session && onboardingComplete === false && !inOnboardingGroup) {
      router.replace('/(onboarding)');
    } else if (session && onboardingComplete === true && (inAuthGroup || inOnboardingGroup)) {
      router.replace('/(tabs)');
    }
  }, [session, loading, onboardingComplete, segments]);

  return <Slot />;
}
```
